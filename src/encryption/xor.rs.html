1//! Payload Encryption -XOR
2//! XOR encryption is the simplest to use and lightest to implement, which makes it a popular choice for malwares.
3//! It is faster than most methods since its a simple CPU operation and needs no libs like AES or RC4 do.
4//! Additionally, it is a bidirectional algorithm, so we have same way to encrypt and decrypt.
5//! This does mean that if someone recognises its XOR, then they can pass the encrypted payload right back
6//! to the routine to get it back as decrypted when reversing.
7
8/// This code snippet below shows a basic encryption function which xors each byte of the shellcode
9/// with a 1 byte key.
10pub fn naieve_xor_by_one_key(shellcode: &mut [u8], key: u8) -> &[u8] {
11    for i in 0..shellcode.len() {
12        shellcode[i] = shellcode[i] ^ key
13    }
14    shellcode
15}
16
17/// There are better ways of securing the encryption key though.
18/// Some tools and security solutions can brute the key which will expose the shellcode.
19/// Since u8 has a very small space for guesswork, so brute methods are easy.
20/// To make the guesswork harder, we can have some minor changes, like using the position
21/// of the shellcode's byte, added to the key so we cannot simply bruteforce it and keyspace is larger.
22pub fn xor_by_ikeys(shellcode: &mut [u8], key: u8) -> &[u8] {
23    for i in 0..shellcode.len() {
24        shellcode[i] = shellcode[i] ^ (key + i as u8)
25    }
26    shellcode
27}
28
29/// This would be a further hardened version. We perform the same encryption process with a single key
30/// but we iterate through bytes of key as well, so we have a larger key-space for encryption.
31/// This way there isn't a single byte of key to brute but "bytes in key" number of keys to bruteforce.
32pub fn xor_by_inputkeys<'a>(shellcode: &'a mut [u8], key: &[u8]) -> &'a [u8] {
33    let mut keyposition = 0;
34    for i in 0..shellcode.len() {
35        if keyposition >= key.len() {
36            keyposition = 0;
37        }
38        shellcode[i] = shellcode[i] ^ key[keyposition];
39        keyposition += 1;
40    }
41
42    shellcode
43}

45#[cfg(test)]
46mod tests {
47    use super::*;
48
49    #[test]
50    fn test_naieve_xor_by_one_key_does() {
51        let mut input = [1, 2, 3, 4, 5, 6, 7, 8];
52        let key = 123;
53        let expected_out = [122, 121, 120, 127, 126, 125, 124, 115];
54        let out = naieve_xor_by_one_key(&mut input[..], key);
55        assert_eq!(expected_out, out);
56    }
57
58    #[test]
59    fn test_xor_by_ikeys() {
60        let mut input = [1, 2, 3, 4, 5, 6, 7, 8];
61        let key = 123;
62        let expected_out = [122, 126, 126, 122, 122, 134, 134, 138];
63        let out = xor_by_ikeys(&mut input[..], key);
64        assert_eq!(expected_out, out);
65    }
66
67    #[test]
68    pub fn test_xor_by_inputkeys() {

70        let key = [10, 20, 30, 40, 50, 60, 70];
71        let expected_out = [11, 22, 29, 44, 55, 58, 65, 2];
72        let out = xor_by_inputkeys(&mut input[..], &key[..]);
73        assert_eq!(expected_out, out);
